183//Function that receives and process the received data from serial port
184//Data is received and accumulated until the complete message is formed
185//Depending on the Tool and device state, each message is procesed differently
186//To prevent processing failure during GPIO deactivation, while it is in progress,
187//this function only reads and clears serial port buffer
188void MainWindow::readData()
189{
190 if(deactivate_GPIO==1){
191 QByteArray temporal_data = p_serie->readAll();
192 return;
193 }
194 if(datareception_enabled==1)
195 {
196 if(operation_mode==0){
197 if (!serial_message.endsWith("\n")){
198 const QByteArray data = p_serie->readAll();
199 serial_message.append(data);
200 ui->p_console->putData(data);
201 }
202 else{
203 distance_analysis(serial_message);
204 serial_message.clear();
205 }
206 }
207 else{
208 if (!position_message.endsWith("\n")){
209 const QByteArray data = p_serie->readAll();
210 position_message.append(data);
211 ui->p_console->putData(data);
212 }
213 else{
214 position_analysis(position_message);
215 position_message.clear();
216 }
217 }
218 }
219 else if(receive_initial_info==1)
220 {
221 if (!info_message.contains(">")){
222 const QByteArray data = p_serie->readAll();
223 info_message.append(data);
224 ui->p_console->putData(data);
225 }
226 if(info_message.contains(">") and (info_message.contains("tn") or info_message.contains("ani") or info_message.contains("an"))){
227 receive_initial_info=0;
228 receive_info();
229 on_pushButton_distancemode_clicked();
230 }
231
232 }
233
234 else{
235 const QByteArray data = p_serie->readAll();
236 ui->p_console->putData(data);
237 }
238}
